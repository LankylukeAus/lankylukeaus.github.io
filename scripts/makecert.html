Generates local TLS certificates for internal services.

Note Live script from my home-lab server. Tokens, IDs, phone numbers and other secrets have been replaced with placeholders like <WHATSAPP_GROUP_ID> — everything else is the real, running code. 
#!/bin/bash

# Variables
DOMAIN="lanky.com"
KEY_FILE="${DOMAIN}.key"
CSR_FILE="${DOMAIN}.csr"
CRT_FILE="${DOMAIN}.crt"
DAYS_VALID=365

echo "Generating private key for $DOMAIN..."
openssl genrsa -out "$KEY_FILE" 2048

echo "Creating Certificate Signing Request (CSR)..."
openssl req -new -key "$KEY_FILE" -out "$CSR_FILE" \
  -subj "/C=AU/ST=YourState/L=YourCity/O=YourOrg/OU=IT Department/CN=$DOMAIN/emailAddress=admin@$DOMAIN"

echo "Generating self-signed certificate..."
openssl x509 -req -days $DAYS_VALID -in "$CSR_FILE" -signkey "$KEY_FILE" -out "$CRT_FILE"

echo "Done! Files generated:"
echo "Private Key: $KEY_FILE"
echo "Certificate: $CRT_FILE"
echo "CSR: $CSR_FILE"
